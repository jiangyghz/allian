public void ProcessRequest(HttpContext context)
        {
            ////在这里添加保存上传文件的代码

            //输出信息,以便Flash能触发onUploadComplete事件.
            //context.Response.Write("ok");

            context.Response.ContentType = "text/plain";
            context.Response.Charset = "utf-8";

            if (context.Request.Files.Count == 0) return;
            HttpPostedFile file = context.Request.Files["Filedata"];

            string filePath;
            FileInfo fileInfo = new FileInfo(file.FileName);
            filePath = Frame.Base.GlobalSetting.TempPath;
            var folder = context.Request.Params["folder"];
            if (!string.IsNullOrEmpty(folder))
            {
                filePath =context.Server.MapPath(folder);
            }
            if (!Directory.Exists(filePath))
            {
                Directory.CreateDirectory(filePath);
            }   
            string newFileName = Guid.NewGuid().ToString() + fileInfo.Extension;

            file.SaveAs(filePath + "\\" + newFileName);
            
 
            //输出文件路径
            context.Response.Write(newFileName);
        }
